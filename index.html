<input type="password" id="password">
<button onclick="togglePassword()">Show</button>

<script>
  function togglePassword() {
    const pwd = document.getElementById('password');
    const btn = event.target;
    if (pwd.type === 'password') {
      pwd.type = 'text';
      btn.innerText = 'Hide';
    } else {
      pwd.type = 'password';
      btn.innerText = 'Show';
    }
  }
</script>
